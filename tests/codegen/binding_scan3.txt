--- TITLE ---
fmt.scan3<A, B, C> — three-value stdin read returning Result<(A, B, C), error>, lowered to the tideScan3 helper (one fmt.Scan of three pointers folded into a 3-tuple Ok); the Ok payload destructures through the tuple-in-variant-payload match path.

--- INPUT ---
import fmt

func main() {
  match fmt.scan3<int, int, int>() {
    Err(_)        => fmt.println("eof"),
    Ok((a, b, c)) => fmt.println(a + b + c),
  }
}

--- GO ---
package main

import "fmt"

type Result[T any, E any] struct {
	Tag uint8
	V   T
	E   E
}

func ResultOk[T any, E any](value T) Result[T, E] {
	return Result[T, E]{Tag: 0, V: value}
}
func ResultErr[T any, E any](err E) Result[T, E] {
	return Result[T, E]{Tag: 1, E: err}
}
func tideScan3[A any, B any, C any]() Result[struct {
	_0 A
	_1 B
	_2 C
}, error] {
	var a A
	var b B
	var c C
	if _, err := fmt.Scan(&a, &b, &c); err != nil {
		return ResultErr[struct {
			_0 A
			_1 B
			_2 C
		}, error](err)
	}
	return ResultOk[struct {
		_0 A
		_1 B
		_2 C
	}, error](struct {
		_0 A
		_1 B
		_2 C
	}{a, b, c})
}
func main() {
	__tide_match_1 := tideScan3[int, int, int]()
	switch __tide_match_1.Tag {
	case 1:
		fmt.Println("eof")
	case 0:
		a := __tide_match_1.V._0
		b := __tide_match_1.V._1
		c := __tide_match_1.V._2
		fmt.Println(a + b + c)
	}
}
