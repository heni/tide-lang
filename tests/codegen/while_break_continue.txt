--- TITLE ---
while loop with continue and break — lowers to Go condition-only for

--- INPUT ---
import fmt

func main() {
  var i = 0
  while i < 10 {
    if i == 3 {
      i += 1
      continue
    }
    if i == 7 {
      break
    }
    fmt.println(i)
    i += 1
  }
}

--- GO ---
package main

import "fmt"

func main() {
	var i = 0
	for i < 10 {
		if i == 3 {
			i = i + 1
			continue
		}
		if i == 7 {
			break
		}
		fmt.Println(i)
		i = i + 1
	}
}
