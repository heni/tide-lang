--- TITLE ---
s.copy() — fresh-backing slice clone via append(s[:0:0], s...)

--- INPUT ---
import fmt

func main() {
  let xs = [1, 2, 3]
  let ys = xs.copy()
  fmt.println(ys.len(), xs.len())
}

--- GO ---
package main

import "fmt"

func main() {
	var xs = []int{1, 2, 3}
	var ys = append(xs[:0:0], xs...)
	fmt.Println(len(ys), len(xs))
}

--- STDOUT ---
3 3

--- EXIT ---
0
