--- TITLE ---
Bare field ident in a method body emits t.<field>; a shadowing local does not

--- INPUT ---
import fmt

class Point {
  let x: int
  let y: int
  sum(): int { return x + y }
  shifted(d: int): int {
    let x = 100
    return x + y
  }
}

func main() {
  let p = Point{ x: 3, y: 4 }
  fmt.println("sum", p.sum())
  fmt.println("shadowed", p.shifted(0))
}

--- GO ---
package main

import "fmt"

type Point struct {
	X int `json:"x"`
	Y int `json:"y"`
}

func (t *Point) sum() int {
	return t.X + t.Y
}
func (t *Point) shifted(d int) int {
	var x = 100
	return x + t.Y
}
func main() {
	var p = &Point{X: 3, Y: 4}
	fmt.Println("sum", p.sum())
	fmt.Println("shadowed", p.shifted(0))
}

--- STDOUT ---
sum 7
shadowed 104

--- EXIT ---
0
