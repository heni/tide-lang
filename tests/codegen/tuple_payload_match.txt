--- TITLE ---
Tuple pattern nested in a variant payload — `Some((a, b))` binds positionally

--- INPUT ---
import fmt

func main() {
  let o: Option<(int, int)> = Some((1, 2))
  match o {
    Some((a, b)) => fmt.println("pair:", a, b),
    None         => fmt.println("none"),
  }
}

--- GO ---
package main

import "fmt"

type Option[T any] struct {
	Tag uint8
	V   T
}

func OptionSome[T any](value T) Option[T] {
	return Option[T]{Tag: 1, V: value}
}
func OptionNone[T any]() Option[T] {
	return Option[T]{Tag: 0}
}
func main() {
	var o Option[struct {
		_0 int
		_1 int
	}] = OptionSome[struct {
		_0 int
		_1 int
	}](struct {
		_0 int
		_1 int
	}{_0: 1, _1: 2})
	__tide_match_1 := o
	switch __tide_match_1.Tag {
	case 1:
		a := __tide_match_1.V._0
		b := __tide_match_1.V._1
		fmt.Println("pair:", a, b)
	case 0:
		fmt.Println("none")
	}
}

--- STDOUT ---
