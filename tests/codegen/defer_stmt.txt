DeferStmt — function-scoped LIFO; lowers to Go `defer` (G27)

--- INPUT ---
import fmt

func main() {
  fmt.println("first")
  defer fmt.println("defer 1")
  fmt.println("second")
  defer (func() { fmt.println("defer 2") })()
}

--- GO ---
package main

import "fmt"

func main() {
	fmt.Println("first")
	defer fmt.Println("defer 1")
	fmt.Println("second")
	defer (func() {
		fmt.Println("defer 2")
	})()
}

--- STDOUT ---
first
second
defer 2
defer 1
