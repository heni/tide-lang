--- TITLE ---
fmt.scan2<A, B> — multi-value stdin read returning Result<(A, B), error>, lowered to the tideScan2 helper (one fmt.Scan of two pointers folded into a tuple Ok); the Ok payload destructures through the tuple-in-variant-payload match path.

--- INPUT ---
import fmt

func main() {
  match fmt.scan2<int, int>() {
    Err(_)     => fmt.println("eof"),
    Ok((a, b)) => fmt.println(a + b),
  }
}

--- GO ---
package main

import "fmt"

type Result[T any, E any] struct {
	Tag uint8
	V   T
	E   E
}

func ResultOk[T any, E any](value T) Result[T, E] {
	return Result[T, E]{Tag: 0, V: value}
}
func ResultErr[T any, E any](err E) Result[T, E] {
	return Result[T, E]{Tag: 1, E: err}
}
func tideScan2[A any, B any]() Result[struct {
	_0 A
	_1 B
}, error] {
	var a A
	var b B
	if _, err := fmt.Scan(&a, &b); err != nil {
		return ResultErr[struct {
			_0 A
			_1 B
		}, error](err)
	}
	return ResultOk[struct {
		_0 A
		_1 B
	}, error](struct {
		_0 A
		_1 B
	}{a, b})
}
func main() {
	__tide_match_1 := tideScan2[int, int]()
	switch __tide_match_1.Tag {
	case 1:
		fmt.Println("eof")
	case 0:
		a := __tide_match_1.V._0
		b := __tide_match_1.V._1
		fmt.Println(a + b)
	}
}
