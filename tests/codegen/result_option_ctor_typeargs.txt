--- TITLE ---
Result/Option constructors stamp explicit type args from the expected type (return + typed let)

--- INPUT ---
import fmt

func classify(n: int): Result<int, string> {
  if n < 0 {
    return Err("negative")
  }
  return Ok(n * 2)
}

func describe(n: int): Option<int> {
  if n == 0 {
    return None()
  }
  return Some(n)
}

func main() {
  let bonus: Result<int, string> = Ok(99)
  match classify(7) {
    Ok(v)  => fmt.println("ok", v),
    Err(e) => fmt.println("err", e),
  }
  match describe(0) {
    Some(v) => fmt.println("some", v),
    None    => fmt.println("none"),
  }
  match bonus {
    Ok(v)  => fmt.println("bonus", v),
    Err(e) => fmt.println("err", e),
  }
}

--- GO ---
package main

import "fmt"

type Option[T any] struct {
	Tag uint8
	V   T
}

func OptionSome[T any](value T) Option[T] {
	return Option[T]{Tag: 1, V: value}
}
func OptionNone[T any]() Option[T] {
	return Option[T]{Tag: 0}
}

type Result[T any, E any] struct {
	Tag uint8
	V   T
	E   E
}

func ResultOk[T any, E any](value T) Result[T, E] {
	return Result[T, E]{Tag: 0, V: value}
}
func ResultErr[T any, E any](err E) Result[T, E] {
	return Result[T, E]{Tag: 1, E: err}
}
func classify(n int) Result[int, string] {
	if n < 0 {
		return ResultErr[int, string]("negative")
	}
	return ResultOk[int, string](n * 2)
}
func describe(n int) Option[int] {
	if n == 0 {
		return OptionNone[int]()
	}
	return OptionSome[int](n)
}
func main() {
	var bonus Result[int, string] = ResultOk[int, string](99)
	__tide_match_1 := classify(7)
	switch __tide_match_1.Tag {
	case 0:
		v := __tide_match_1.V
		fmt.Println("ok", v)
	case 1:
		e := __tide_match_1.E
		fmt.Println("err", e)
	}
	__tide_match_2 := describe(0)
	switch __tide_match_2.Tag {
	case 1:
		v := __tide_match_2.V
		fmt.Println("some", v)
	case 0:
		fmt.Println("none")
	}
	__tide_match_3 := bonus
	switch __tide_match_3.Tag {
	case 0:
		v := __tide_match_3.V
		fmt.Println("bonus", v)
	case 1:
		e := __tide_match_3.E
		fmt.Println("err", e)
	}
}
