--- TITLE ---
Record type → Go struct; record literal → struct literal; noBrace header

--- INPUT ---
import fmt

type Point = { x: int, y: int }

func main() {
  let p = Point { x: 3, y: 4 }
  if p.x > 0 {
    fmt.println(p.x, p.y)
  }
}

--- GO ---
package main

import "fmt"

type Point struct {
	X int `json:"x"`
	Y int `json:"y"`
}

func main() {
	var p = Point{X: 3, Y: 4}
	if p.X > 0 {
		fmt.Println(p.X, p.Y)
	}
}

--- STDOUT ---
3 4

0
