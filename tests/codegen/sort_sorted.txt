--- TITLE ---
sort.sorted(s, less): the comparator binding lowers to the inline tideSorted helper (copy + sort.SliceStable, returning a new slice), and the untyped short-closure params are stamped from the slice element type (T-Closure).

--- INPUT ---
import fmt
import sort

func main() {
  let xs = [3, 1, 2]
  let ys = sort.sorted(xs, (a, b) => a < b)
  for y in ys {
    fmt.println(y)
  }
}

--- GO ---
package main

import (
	"fmt"
	"sort"
)

func tideSorted[T any](s []T, less func(T, T) bool) []T {
	out := make([]T, len(s))
	copy(out, s)
	sort.SliceStable(out, func(i, j int) bool { return less(out[i], out[j]) })
	return out
}
func main() {
	var xs = []int{3, 1, 2}
	var ys = tideSorted(xs, func(a int, b int) bool {
		return a < b
	})
	for _, y := range ys {
		fmt.Println(y)
	}
}
